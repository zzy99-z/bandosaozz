// Minimal Node/Express proxy for /api/generate // Requirements: Node 18+ (has global fetch), or install node-fetch for older Node. const express = require('express'); const path = require('path'); const app = express(); const PORT = process.env.PORT || 3000; app.use(express.json()); app.use(express.static(path.join(__dirname, 'public'))); // serve index.html from ./public app.post('/api/generate', async (req, res) => { try { const { name, dob, tob } = req.body || {}; if (!name || !dob) return res.status(400).json({ error: 'Missing name or dob' }); const apiKey = process.env.GEN_API_KEY; if (!apiKey) return res.status(500).json({ error: 'Server missing GEN_API_KEY env variable' }); // Build prompt on server to avoid trusting client inputs alone const prompt = `Bạn là nhà chiêm tinh học chuyên nghiệp của BẠN ĐỒ SAO Zz.\n` + `Hãy giải mã bản đồ sao cho: ${name}, sinh ngày: ${dob}, giờ sinh: ${tob || 'Không rõ giờ sinh'}.\n` + `Phân tích chi tiết các mục sau: Tuổi thơ, Gia đình, Sức khỏe, Sự nghiệp, Tiền bạc, Nhà cửa, Cha mẹ, Bạn bè, Anh em, Tình duyên, Hôn nhân, Con cái.\n` + `Yêu cầu: Trả về kết quả bằng tiếng Việt, định dạng HTML đẹp mắt (dùng

cho tiêu đề mục,

cho nội dung). Văn phong huyền bí, lôi cuốn.`; const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${apiKey}`; const body = { contents: [{ parts: [{ text: prompt }] }] }; const apiResp = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(body) }); if (!apiResp.ok) { const text = await apiResp.text().catch(() => ''); console.error('Generative API error', apiResp.status, text); return res.status(502).json({ error: 'Upstream API error', status: apiResp.status, text }); } const data = await apiResp.json(); // Extract content from likely response shapes const content = data.content ?? data.text ?? data.result ?? data.candidates?.[0]?.content?.parts?.[0]?.text ?? data.candidates?.[0]?.content?.[0]?.text ?? data.output?.[0]?.content?.[0]?.text ?? null; if (!content) { console.warn('Unexpected API response shape', data); return res.status(502).json({ error: 'Unexpected upstream response' }); } // Return a simple JSON containing the HTML content res.json({ content }); } catch (err) { console.error('Server error /api/generate', err); res.status(500).json({ error: 'Internal server error' }); } }); app.listen(PORT, () => { console.log(`Server listening on ${PORT}`); });
